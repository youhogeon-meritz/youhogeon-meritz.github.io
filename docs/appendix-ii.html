A2

부록 II

수정을 위해 Jlox의 구문 트리 클래스를 자동으로 생성하기 위해 구축한 작은 스크립트에서 생성된 코드를 소개합니다.

A2 . 1표현.

표현식은 "표현 코드"에서 처음으로 소개된 구문 트리 노드입니다. 메인 Expr 클래스는 특정 표현식 유형에 대해 디스패치하는 데 사용되는 방문자 인터페이스를 정의하며, 다른 표현식 하위 클래스는 중첩된 클래스로 포함됩니다.

lox/Expr.java
새 파일 만들기
package com.craftinginterpreters.lox;

import java.util.List;

abstract class Expr {
  interface Visitor<R> {
    R visitAssignExpr(Assign expr);
    R visitBinaryExpr(Binary expr);
    R visitCallExpr(Call expr);
    R visitGetExpr(Get expr);
    R visitGroupingExpr(Grouping expr);
    R visitLiteralExpr(Literal expr);
    R visitLogicalExpr(Logical expr);
    R visitSetExpr(Set expr);
    R visitSuperExpr(Super expr);
    R visitThisExpr(This expr);
    R visitUnaryExpr(Unary expr);
    R visitVariableExpr(Variable expr);
  }

  // Nested Expr classes here...

  abstract <R> R accept(Visitor<R> visitor);
}

A2 . 2진술들

문은 표현식과 무관하게 구문 트리 노드의 두 번째 계층을 형성합니다. "진술과 진술"에서 처음 몇 가지를 추가합니다.

package com.craftinginterpreters.lox;

import java.util.List;

abstract class Stmt {
  interface Visitor<R> {
    R visitBlockStmt(Block stmt);
    R visitClassStmt(Class stmt);
    R visitExpressionStmt(Expression stmt);
    R visitFunctionStmt(Function stmt);
    R visitIfStmt(If stmt);
    R visitPrintStmt(Print stmt);
    R visitReturnStmt(Return stmt);
    R visitVarStmt(Var stmt);
    R visitWhileStmt(While stmt);
  }

  // Nested Stmt classes here...

  abstract <R> R accept(Visitor<R> visitor);
}
